                  Driver emits verbatim 'manual deploy required' token; Pester asserts.

      0 — all 4 Describe blocks green

.PARAMETER WeaselPath
    Override for Weasel install dir; default = probe Program Files\Rime\weasel-*.

.PARAMETER RepoRoot
    Repo root (where schema/ lives). Default: parent of script's dir.

jobs:

    timeout-minutes: 30

        uses: actions/checkout@v4

      - name: Show runner identity

        # safe on the very first job (clean runner) where the dirs don't exist;

        shell: pwsh
        run: |
          Remove-Item -Recurse -Force -ErrorAction SilentlyContinue $env:APPDATA\Rime, "$env:LOCALAPPDATA\Rime"
          Write-Host "Clean-slate Remove-Item completed (errors silently ignored if dirs absent on first job)"

        # on github-hosted (no desktop session). For CI we use --silent and
        # accept the documented hang risk, capped by Start-Process -Wait
        # -Timeout 300s. If winget hangs, kill it RED rather than letting the
        # GHA runner sit at the 6-hour default timeout.

        # Homebrew --no-quarantine d4ba9db incident. If winget hangs / package
        # name drifts at execute time, escalate to gsd-checker — do NOT silently
        # mask with `|| true`.
        shell: pwsh
        run: |
          $proc = Start-Process -FilePath winget `
                                -ArgumentList 'install','--id','Rime.Weasel',
                                              '--silent','--accept-source-agreements',
                                              '--accept-package-agreements' `
                                -PassThru -NoNewWindow

            exit 1
          }

            Write-Error "ERROR: winget install Rime.Weasel exited non-zero ($($proc.ExitCode)). Check whether Rime.Weasel package id is still valid (Win equivalent of Phase 2's Homebrew --no-quarantine deprecation incident)."
            exit 1
          }
          Write-Host "winget install Rime.Weasel succeeded ($($proc.ExitCode))"

      - name: Resolve Weasel install path (versioned subdir)
        # winget installs Rime.Weasel to a VERSIONED subdir (e.g.

              $versioned = Get-ChildItem $parent -Directory -Filter 'weasel-*' -ErrorAction SilentlyContinue |
                           Sort-Object Name -Descending | Select-Object -First 1

              exit 1

          Write-Host "rime.dll present at $WeaselDll"

        # - Schema files installed + dict.yaml SHA match (Describe 2)
        # - GUI-skip 'manual deploy required' token (Describe 3)
        # - Authenticode NotSigned regression guard (Describe 4)

        run: |
          Import-Module Pester -MinimumVersion 5.0.0
          $config = New-PesterConfiguration
          $config.Run.Path = 'tests/test_install_e2e_win.ps1'
          $config.Output.Verbosity = 'Detailed'
          $config.Run.Exit = $true
          $config.Run.PassThru = $true
          $result = Invoke-Pester -Configuration $config
          if ($result.Result -ne 'Passed') {
              Write-Error "ERROR: Pester driver reported $($result.FailedCount) failures."
              exit 1
          }
          Write-Host "Pester driver: all $($result.PassedCount) tests passed"

      - name: Assert manual-deploy token surfaced (CP-4 belt-and-suspenders)
        # Re-run install-windows.ps1 capturing stdout, then grep for the
        # verbatim 'manual deploy required' token. Mirrors Phase 2 Plan 02-01's

        # not silent skip. Driver itself emits this token (D5); install-
        # windows.ps1 emits 'Auto-deploy skipped' (also asserted, defense-in-
        # depth).
        shell: pwsh
        env:
          SMOODLE_AUTO_DEPLOY: "0"
          SMOODLE_NONINTERACTIVE: "1"
        run: |
          $out = & powershell -NoProfile -ExecutionPolicy Bypass -File scripts/install-windows.ps1 2>&1 | Out-String
          Write-Host $out
          if ($out -notmatch 'Auto-deploy skipped') {
              Write-Error "ERROR: 'Auto-deploy skipped' marker missing - SMOODLE_AUTO_DEPLOY=0 env-override regressed in install-windows.ps1"
              exit 1
          }
          Write-Host "[OK] 'Auto-deploy skipped' marker present (CP-4 invariant: WeaselDeployer never invoked)"

      - name: Verify dict.yaml SHA in destination matches repo source
        # ROADMAP SC #1 - independent re-check (driver also does this; this step

        # Mirrors Phase 2 Plan 02-01's last step.

              Write-Error "ERROR: dict.yaml SHA mismatch between schema/ source and %APPDATA%\Rime\ destination"

          }
          Write-Host "[OK] dict.yaml SHA match"
